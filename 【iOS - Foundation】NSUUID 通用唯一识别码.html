NSUUID 通用唯一识别码

  • 1、创建
  • 2、信息

NSUUID : NSObject

1、创建

    // 随机
    NSUUID *uuid = [NSUUID UUID];
    
    // 字符串必须按格式
    NSUUID *uuid2 = [[NSUUID alloc] initWithUUIDString:@"68753A44-4D6F-1226-9C60-0050E4C00067"];

    Byte byte[] = {1,2,3,4};    
    NSUUID *uuid3 = [[NSUUID alloc] initWithUUIDBytes:byte];

2、信息

    // 转字符串
    NSString *uuidString = uuid.UUIDString;
    
    // 不明所以
    [uuid2 getUUIDBytes:byte];
